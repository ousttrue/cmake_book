add_subdirectory, subdirs 例

# project
CMAKE_MINIMUM_REQUIRED(VERSION 3.20.0)
PROJECT(hello VERSION 0.1.0)

ADD_SUBDIRECTORY(hello)
ADD_SUBDIRECTORY(src)

# 同じ

SUBDIRS(hello src)
